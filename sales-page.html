Developer Paradise 🛠️

Extensive hooks, filters, and APIs for complete customization

// Award points when user reshares an activity
add_action( 'bp_share_user_reshared_activity', function( $user_id, $reshare_type, $original_activity, $new_activity_id ) {
    // Award 10 points for resharing
    if ( function_exists( 'mycred_add' ) ) {
        mycred_add( 'activity_reshare', $user_id, 10, 'Reshared an activity' );
    }
    
    // Trigger achievement check
    do_action( 'gamipress_trigger_event', 'bp_reshare_activity', $user_id );
}, 10, 4 );

// Customize share tracking parameters
add_filter( 'bp_share_tracking_parameters', function( $params, $activity_id, $service ) {
    $params['custom_campaign'] = 'summer_2024';
    $params['user_segment'] = get_user_meta( get_current_user_id(), 'segment', true );
    return $params;
}, 10, 3 );

// Add custom social service
add_filter( 'bp_share_services', function( $services ) {
    $services['custom_platform'] = array(
        'name' => 'Custom Platform',
        'icon' => 'fa-custom',
        'url'  => 'https://custom.com/share?url={url}&text={title}'
    );
    return $services;
});
